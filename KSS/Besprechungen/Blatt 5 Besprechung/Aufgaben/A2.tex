\begin{align*}
&X_k=\begin{cases}
1&K_1<K_2<K_3<\ldots<K_n\\
0&\text{sonst}
\end{cases}\\
&\overset{2.6.5}{\rightarrow}E(X)=\sum_{k=1}^{n}P(X_k=1)\\
&P(X_k=1)=\frac{1}{(k-1)!}\\
&\rightarrow\sum_{k=1}^{n}\frac{1}{(k-1)!}=\sum_{k=0}^{n-1}\frac{ 1 }{ k! } \overset{n \rightarrow \infty}{\rightarrow }e\\ 
\end{align*}
