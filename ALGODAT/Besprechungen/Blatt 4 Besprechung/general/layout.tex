%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% LAYOUT SETTING - using fancyhdr
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\submissiontitle}[1]{
	\pagestyle{fancy}
	\setlength{\headheight}{14.5pt}
	\fancyhead[R]{\thepage}
	\fancyhead[L]{#1}
	\fancyfoot{}
	\title{#1}
}
